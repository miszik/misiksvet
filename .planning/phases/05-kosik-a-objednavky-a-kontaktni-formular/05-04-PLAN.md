node -e "
const fs = require('fs');

const checks = [
  // Soubory musi existovat
  { label: 'js/cart.js existuje', fn: () => fs.existsSync('js/cart.js') },
  { label: 'js/order.js existuje', fn: () => fs.existsSync('js/order.js') },

  // products.js ma type pole
  { label: 'products.js ma type:uterka (3x)', fn: () => {
    const c = fs.readFileSync('js/products.js', 'utf8');
    return (c.match(/type: 'uterka'/g) || []).length === 3;
  }},
  { label: 'products.js ma type:zastera (1x)', fn: () => {
    const c = fs.readFileSync('js/products.js', 'utf8');
    return (c.match(/type: 'zastera'/g) || []).length === 1;
  }},

  // cart.js ma klic funkce
  { label: 'cart.js: calcPromo filtruje type uterka', fn: () =>
    fs.readFileSync('js/cart.js', 'utf8').includes(\"type === 'uterka'\") },
  { label: 'cart.js: Math.floor pro promo', fn: () =>
    fs.readFileSync('js/cart.js', 'utf8').includes('Math.floor') },
  { label: 'cart.js: try/catch kolem localStorage', fn: () =>
    fs.readFileSync('js/cart.js', 'utf8').includes('try {') },

  // order.js bezpecnostni kriteria
  { label: 'order.js: e.preventDefault() je prvni', fn: () => {
    const c = fs.readFileSync('js/order.js', 'utf8');
    const submitIdx = c.indexOf('async function handleFormSubmit');
    const preventIdx = c.indexOf('e.preventDefault()', submitIdx);
    const validateIdx = c.indexOf('validateForm()', submitIdx);
    return preventIdx < validateIdx;
  }},
  { label: 'order.js: submitBtn.disabled = true pred send', fn: () =>
    fs.readFileSync('js/order.js', 'utf8').includes('submitBtn.disabled = true') },
  { label: 'order.js: emailjs.send (ne sendForm)', fn: () => {
    const c = fs.readFileSync('js/order.js', 'utf8');
    return c.includes('emailjs.send(') && !c.includes('emailjs.sendForm');
  }},
  { label: 'order.js: clearCart() po uspechu', fn: () =>
    fs.readFileSync('js/order.js', 'utf8').includes('clearCart()') },
  { label: 'order.js: prazdny kosik validace', fn: () =>
    fs.readFileSync('js/order.js', 'utf8').includes('Košík je prázdný') },

  // index.html script order
  { label: 'index.html: packeta CDN pred cart.js', fn: () => {
    const h = fs.readFileSync('index.html', 'utf8');
    return h.indexOf('packeta.com/v6') < h.indexOf('js/cart.js');
  }},
  { label: 'index.html: cart.js pred order.js pred main.js', fn: () => {
    const h = fs.readFileSync('index.html', 'utf8');
    return h.indexOf('js/cart.js') < h.indexOf('js/order.js') &&
           h.indexOf('js/order.js') < h.indexOf('js/main.js');
  }},

  // main.js vola vse
  { label: 'main.js: initCart() a initOrderForm()', fn: () => {
    const c = fs.readFileSync('js/main.js', 'utf8');
    return c.includes('initCart()') && c.includes('initOrderForm()');
  }},
];

let passed = 0, failed = 0;
checks.forEach(({ label, fn }) => {
  try {
    const ok = fn();
    console.log((ok ? '  PASS' : '  FAIL') + ': ' + label);
    if (ok) passed++; else failed++;
  } catch (e) {
    console.log('  ERROR: ' + label + ' — ' + e.message);
    failed++;
  }
});
console.log('');
console.log('Vysledek: ' + passed + '/' + (passed+failed) + ' proslo');
if (failed > 0) process.exit(1);
"
